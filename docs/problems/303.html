def fSigma(bound):
	start = time()
	fDict = dict(zip(range(1, bound+1), [0]*bound))
	total = 0
	for n in xrange(1, bound+1):
		try:
			if fDict[n/10.0] != 0:
				total += fDict[n/10]
				fDict[n] = fDict[n/10]
		except:
			ratio = fbase3(n)/n
			total += ratio
			fDict[n] = ratio
		
		if n%1000 == 0:
