def leap(y):
	if y in [1900, 2000]: return 'FALSE'
	elif y%4 == 0: return 'TRUE'
	else: return 'FALSE'
	
def monthlength(m, y):
	if m in [1, 3, 5, 7, 8, 10, 12]: return 31
	elif m in [4, 6, 9, 11]: return 30
	elif m == 2:
		if leap(y) == 'TRUE': return 29
		else: return 28
