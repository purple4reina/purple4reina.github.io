from itertools import *

def isprime(n):
	if n == 2:
		return True
	elif n < 2 or n%2 == 0:
		return False
	else:
		for i in range(3, int(n**0.5 + 1), 2):
			if n%i == 0: return False
		return True


def primesbelow(h):
	answer = [2]
	for n in range(3, h, 2):
		if isprime(n) == True:
			answer.append(n)
	return answer
