def T(n, k):
    n -= 1
    k -= 1
    return k * (n // k) * ((n + k) // k) // 2 - (n // k) * (k - 1 - (n % k))

def solve(n):
    total = 0
    for k in range(2, n + 1):
        total += T(n, k)
    return total

if __name__ == '__main__':
    import sys
    n = eval(sys.argv[1])
    print(solve(n))
