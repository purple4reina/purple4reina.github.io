from time import time
start = time()

def isprime(n):
	if n == 2:
		return True
	elif n < 2 or n%2 == 0:
		return False
	else:
		for i in range(3, int(n**0.5 + 1), 2):
			if n%i == 0: return False
		return True


primes = []
for n in xrange(10**6):

def nextprime(p):

			ans.append(i-1)
		p = nextprime(p)
	return ans
