Code

modulo = 1000000007

def solve(num):
    total = tri = addr = 0
    while True:
        addr += 1
        tri += addr
        if tri > num:
            break
        div, mod = divmod(num - tri + 1, addr)
        total += div * (div + 1) // 2 * addr + mod * (div + 1)
    return total % modulo

if __name__ == '__main__':
    import sys
    n = eval(sys.argv[1])
    print(solve(n))
